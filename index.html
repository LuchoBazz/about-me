int go(int i, int w) {
if (i < 0 || w <= 0) return 0;
if (dp[i][w] != -1) return dp[i][w];
int res = go(i - 1, w);
if (weights[i] <= w)
res = max(res, values[i] + go(i - 1, w - weights[i]));
return dp[i][w] = res;
}
